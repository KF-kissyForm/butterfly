Textbox 文本输入框组件 Demo

pattern

pattern

通过config参数配置

code
                

                    
                    
                

            
                KISSY.use('gallery/form/1.3/textbox/index', function (S, TextBox) {
                    var textbox = new TextBox.TextBox('#textbox', {
                            pattern: '^\d*$'
                        });
                        textbox.render();

                        textbox.on('invalid', function (ev) {
                            alert('textbox invalid: ' + this.get('validationMessage'));
                        });

                        S.one('#textbox button').on('click', function () {
                            alert(textbox.checkValidity());
                        });
                });

patternMismatchMessage，默认为“输入有误”

通过config参数配置

code
                

                    
                    
                

            
                KISSY.use('gallery/form/1.3/textbox/index', function (S, TextBox) {
                    var textbox = new TextBox.TextBox('#textbox', {
                            patternMismatchMessage: '请输入内容'
                        });
                        textbox.render();

                        textbox.on('invalid', function (ev) {
                            alert('textbox invalid: ' + this.get('validationMessage'));
                        });

                        S.one('#textbox button').on('click', function () {
                            alert(textbox.checkValidity());
                        });
                });
